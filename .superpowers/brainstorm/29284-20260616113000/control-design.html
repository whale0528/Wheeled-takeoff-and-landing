纵向通道：公式不变，数值重算

wd_pitch = 15; kesi_pitch = 0.8; Tr = 1.2;

a24 = -Cma_local * qc * S_ref * c_ref / Iyy;
a25 = -Cmde * qc * S_ref * c_ref / Iyy;
a34 = CLa_local * qc * S_ref / (m * V_control);

Kw_pitch = -2 * kesi_pitch * wd_pitch / a25;
K_alpha = (wd_pitch² - a24) / (2 * kesi_pitch * wd_pitch);
Kn_pitch = -(2.2 * g * wd_pitch²) / (Tr * (wd_pitch² - a24) * V_control * a34);

横航向通道：同样替换

b24 = -Cnβ * qc * S_ref * b_ref / Izz;
b27 = -Cndr * qc * S_ref * b_ref / Izz;
b34 = (D0 - CYβ * qc * S_ref) / (m * V_control);
b37 = -CYdr * qc * S_ref / (m * V_control);

Kw_yaw = -2 * kesi_yaw * wd_yaw / b27;
K_beta = (wd_yaw² - b24) / (2 * kesi_yaw * wd_yaw);
Kn_yaw = (2.2 * g * wd_yaw²) / (Tr_yaw * (wd_yaw² - b24) * V_control * b34);
