组装后的气动模型（Simulink 中查表实现）

CL = K_GE * [CL_table(α) + CLde * K_δe(α) * δe]
  + 1.867 * g_L_TE(δf) % 后缘襟翼
  + 0.01*1.867 * g_L_LE(δs) % 前缘缝翼
CD = CD0 + ΔCD0(M)
  + k_i_eff * (CL - CL_D0_config)^2
  + 0.1316 * g_D_TE(δf) % 襟翼增阻
  + 0.5*0.1316 * g_D_LE(δs) + 0.0372*η_gear
Cm = Cm_table(α) + Cmde * K_δe(α) * δe
  - 0.084*(δf/δf_max + δs/δs_max) + 0.0022*η_gear
